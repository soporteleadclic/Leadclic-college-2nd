trigger CaseDescription on Case (before update) {
    
    Map<Id,Case> caseWithPCase = new Map<Id,Case>(
        [SELECT Id,(SELECT Id,Casenumber FROM Cases) FROM case WHERE Id IN :Trigger.New]);

    for(Case a : Trigger.new) {
        list<String> idhijos= new list<String>();
        if (caseWithPCase.get(a.Id).cases.size() == 0) {
            a.Description='No tiene casos hijos';
        } else {
            for (case b :caseWithPCase.get(a.Id).cases){
                idhijos.add (b.CaseNumber);
                a.Description = 'los casos hijos son' + idhijos;
                System.debug('se actualizo un caso');
            }
        }
    }
}
