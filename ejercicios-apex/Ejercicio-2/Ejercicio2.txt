trigger ChildCaseNumbers on Case (before update) {
    Map<Id, List<String>> ChildCase = new Map<Id, List<String>>();
    
    
    for(Case c :[SELECT Id, ParentId, CaseNumber FROM Case WHERE ParentId IN :trigger.new]){
     
     if(!ChildCase.containsKey(c.ParentId)){
         ChildCase.put(c.ParentId, new List<String>());
          
 }
         ChildCase.get(c.ParentId).add(c.CaseNumber);
     }
 
     
     for(Case casetrigger :trigger.new){
     List<String> numberOfsonCase = ChildCase.get(casetrigger.Id);
 
     if(numberOfsonCase != null){
         // The list is not null, so combine its contents
         casetrigger.Description = String.join(numberOfsonCase, ',');
     }else{
         // The list is null, so use some default text instead of the list
         casetrigger.Description = 'No child cases.';
     }
     
     
     
     }

}
